        虽然已经有很多博友给出了这个算法的Phython实现过程，说实话，利用Phython的实现比C++来的简答，但是不是每位博友都会Phython，所以在这里提供C++的实现过程，以备各位博友的不时之需！


        好的，话不多说。这次主要任务是实现PLA的pocket过程，前面博客提到，当训练样本是线性可分的，我们用PLA可以很快的实现样本的分类。但是如果样本不是线性可分的，那么我们就需要对PLA进行改进。这里采用的是贪心算法，他的思想是把当前最好的分类线保存在口袋中，然后对曲线进行修正得到新的线。如果得到新的线对训练样本的错误率更小，那么我们把这条线保存下来。继续运行程序，直到达到足够的迭代次数。

0.初始化口袋曲线w

{

    1.寻找分类错误点（x,y）

    2.修正错误：w(t+1) = w(t) + y*x

    3.如果w(t+1)对训练样本的错误率比口袋里的w更小，则用w(t+1)替代w

}until(达到足够的迭代次数)


        如果知道训练样本D是线性可分的，则运行PLA比较好，应为PLA速度快（不用判断对所有样本的错误率）

        如果训练样本不是线性可分的（绝大多数情况），则运行pocket，但是pocket得运行速度慢


1.第18题

（1）题意：首先分别从题目中写的两个网址中下载训练样本和测试样本，然后运行pocket算法，每次迭代50次，共运行2000次，计算他对测试样本的平均错误率

（2）实现：

  1. #include<fstream>
  2. #include<iostream>
  3. #include<vector>
  4. #include<algorithm>
  5.  
  6. using namespace std;
  7.  
  8. #define DEMENSION 5  //数据维度
  9.  
  10. //样本结构体
  11. struct record{
  12. 	double x[DEMENSION];
  13. 	int y;
  14. };
  15.  
  16. //读取文件数据
  17. void getData(fstream &datafile,vector<record> &dataset){
  18. 	while(!datafile.eof()){
  19. 		record curRecord;
  20. 		curRecord.x[0] = 1;
  21. 		int i;
  22. 		for(i=1 ; i<DEMENSION ; i++)datafile>>curRecord.x[i];
  23. 		datafile>>curRecord.y;
  24. 		dataset.push_back(curRecord);
  25. 	}
  26. 	datafile.close();
  27. }
  28.  
  29. //计算sign值
  30. int sign(double x){
  31. 	if(x <= 0)return -1;
  32. 	else return 1;
  33. }
  34.  
  35. //计算两个向量内积，判断是否需要修正
  36. double multiply(double *v1, double *v2){
  37. 	int i;
  38. 	double temp = 0.0;
  39. 	for(i = 0; i < DEMENSION; i++)temp += v1[i] * v2[i];
  40. 	return temp;
  41. }
  42.  
  43. //函数重载，计算向量v与整数num的积，用于计算y*x（y为+1或-1，x为向量）
  44. void multiply(double *result,double *v,int num){
  45. 	int i;
  46. 	for(i = 0; i < DEMENSION; i++)result[i] =  num * v[i];
  47. }
  48.  
  49. //计算两向量的和放入result中，用于计算w(i+1)=w(i)+y*x
  50. void add(double *result,double *v1,double *v2){
  51. 	int i;
  52. 	for(i = 0; i < DEMENSION; i++)result[i] = v1[i] + v2[i];
  53. }
  54.  
  55. //计算错误率
  56. double getErrorRate(double *weight,vector<record> dataset){
  57. 	int n = dataset.size();
  58. 	double errorRate= 0.0;
  59. 	int i;
  60. 	for(i=0;i<n;i++)
  61. 		if(sign(multiply(weight,dataset[i].x)) != dataset[i].y)errorRate++;
  62. 	return errorRate/n;
  63. }
  64.  
  65. //口袋PLA算法
  66. void pocketPLA(double *pocketWeights,double *weights,vector<record> trainingSet,int iteration){
  67. 	int index = 0;
  68. 	int iter= 1;
  69. 	int n = trainingSet.size();
  70. 	while(iter < iteration){
  71. 		if(sign(multiply(trainingSet[index].x,weights)) != trainingSet[index].y){
  72. 			double temp[DEMENSION];
  73. 			multiply(temp,trainingSet[index].x,trainingSet[index].y);
  74. 			int i;
  75. 			for(i=0;i<DEMENSION;i++)weights[i] += temp[i];
  76. 	    	if(getErrorRate(weights,trainingSet) < getErrorRate(pocketWeights,trainingSet)){
  77. 				int j;
  78. 				for(j = 0;j<DEMENSION;j++)pocketWeights[j] = weights[j]; 
  79. 			}
  80. 			iter++;
  81. 		}
  82. 		if(index == n-1)index = 0;
  83. 		else index++;
  84. 	}
  85. }
  86.  
  87. void main(){
  88. 	vector<record> trainingSet;
  89. 	vector<record> testSet;
  90. 	fstream datafile1("training_data.txt");
  91. 	fstream datafile2("test_data.txt");
  92. 	if(datafile1.is_open()&&datafile2.is_open()){
  93. 		getData(datafile1,trainingSet);
  94. 		getData(datafile2,testSet);
  95. 	}
  96. 	else{
  97. 		cout<<"can not open file!"<<endl;
  98. 		exit(1);
  99. 	}
  100. 	double weights[DEMENSION],pocketWeights[DEMENSION];
  101. 	
  102. 	double ave_error = 0.0 ;
  103. 	int j;
  104. 	for(j = 0; j < 2000; j++ ){
  105.  
  106. 		random_shuffle(trainingSet.begin(), trainingSet.end());
  107.  
  108. 		int i;
  109. 		for(i=0;i<DEMENSION;i++){  //注意，这里需要初始化！！！不初始化值会乱码，程序出错！！！
  110. 		    weights[i]=0.0;
  111. 		    pocketWeights[i]=0.0;
  112. 		}
  113. 		pocketPLA(pocketWeights,weights,trainingSet,50);
  114. 	    double trainingError = getErrorRate(pocketWeights,trainingSet);
  115.    	    double testError = getErrorRate(pocketWeights,testSet);
  116. 		ave_error += testError;
  117. 		cout<<"第"<<j<<"次实验---"<<"training error:"<<trainingError<<" test error:"<<testError<<endl;
  118.  
  119. 	}
  120. 	cout<<"average error rate:"<<ave_error/2000<<endl;
  121. }

(3)实验结果：个人的计算机计算出来为0.132011


2.第19题

（1）题意：如果不采用贪心算法，而是采用第50次迭代结果为最终曲线，运行2000次，求对测试样本的平均错误率

（2）分析：很简单，只要把pocketPLA函数中

  1. 	if(getErrorRate(weights,trainingSet) < getErrorRate(pocketWeights,trainingSet)){
  2. 				int j;
  3. 				for(j = 0;j<DEMENSION;j++)pocketWeights[j] = weights[j]; 
  4. 			}

这个if判断去掉就可以了

（3）答案：个人计算机结果为0.271567


3.第19题

（1）题意：如果把每次调用口袋算法的迭代次数从50次调为100次，求对测试样本的平均错误率

（2）分析：这个更简单，我写的pocketPLA函数有个参数iteration这个就是迭代次数，在main调用时，把pocketPLA(pocketWeights,weights,trainingSet,50)的50改为100就可以了。由于迭代次数增加，所以错误率应该比18中的错误率小一些。

（3）答案：个人计算机得到的结果为0.114024
