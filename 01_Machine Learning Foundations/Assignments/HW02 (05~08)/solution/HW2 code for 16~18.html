1.第16题

（1）题意：这道题说的是“positive and negative rays”，这种分类方法在老师课堂中已经分析过了，mH（N）=2N。我们在区间[-1，1]上取若干个点（17题为20个），这20个点将[-1，1]分成21个区间，theta可以取21个区间内任意一个，加上s的取值可以为-1或者1，共有21*2=42种组合方式。分别对这42种hyphothesis计算h（x），计算他与y的区别，即E_in，选择这42种组合最小E_in，把这个hyphothesis作为最佳理论，通过他计E_out。

（2）分析：由第一题我们知道加噪声后的计算方式是此式子。对于这道题我们加20%噪声，所以lambda = 0.8,我们只要求mu就可以了。mu的定义是h（x）与f(x)的不同，即错误率。f(x)=s（x）=sign(x)已经在给出，h(x)=s*sign（x-theta）在题中也给出了，所以我们需要根据s和theta分类讨论

        1.s = 1, theta > 0:错误率为theta/2

        2.s=1，theta < 0;错误率为|theta|/2

        3.s=-1,theta > 0:错误率为 （2 - theta）/2

        4.s=-1,theta <0:错误率为（2- | theta |）/2

        综上，s=1 错误率为 |theta|/2;s = -1，错误率为（2-|theta|）/2

        利用一个式子写出来： mu = (s+1)/2 * (|theta|/2) -  (s-1)/2 * ((2-|theta|)/2)

       最后 E_out = mu * lambda + (1 - lambda) * (1 - mu)，lambda = 0.8,mu带入可以得到答案

（3）答案：0.5+0.3*s*(|theta| - 1)


2.第17，18题

 （1）题意：

        第17题的意思是在[-1,1]种取20个点，分隔为21个区间作为theta的取值区间，每种分类有42个hyphothesis，枚举所有可能情况找到使E_in最小的hyphothesis，记录最小E_in

        第18题的意思是在17题得到的最佳hyphothesis的基础上，利用第16题的公式计算E_out.

（2）实现代码

  1. #include<iostream>
  2. #include<stdlib.h>
  3. #include<vector>
  4. #include<algorithm>
  5. #include<math.h>
  6. using namespace std;
  7.  
  8. #define DATASIZE 20 //定义[-1,1]的内点的数目
  9.  
  10. //训练样本结构体
  11. struct record{
  12. 	double x;
  13. 	int y;
  14. };
  15.  
  16. //hyphothesis的结构体，s为+1或-1，theta在20个点分隔的21个区间取值
  17. struct hyphothesis{
  18. 	int s;
  19. 	double theta;
  20. };
  21.  
  22. //sign函数
  23. int sign(double x){
  24. 	if(x <= 0)return -1;
  25. 	else return 1;
  26. }
  27.  
  28. //随机在[-1,1]内生成DATASIZE个点的x，并计算对应的y
  29. void getRandData(vector<record> &trainingData){
  30. 	int i;
  31. 	for(i = 0; i < DATASIZE; i++){
  32. 		record temp;
  33. 		temp.x = 2.0 * rand() / double(RAND_MAX) - 1.0;
  34. 		temp.y = sign(temp.x);
  35. 		trainingData.push_back(temp);
  36. 	}
  37. }
  38.  
  39. //添加噪声，即把20%的点的y值正负号颠倒。这里的20%可以通过随机方法得到[0,1]的数，若小于0.2则认为这个点加噪声
  40. void getNoise(vector<record> &trainingData){
  41. 	int i;
  42. 	for(i = 0; i < DATASIZE; i++){
  43. 		double randnum = rand() / double(RAND_MAX);
  44. 		if(randnum < 0.2)
  45. 			trainingData[i].y = -1 * trainingData[i].y;
  46. 	}
  47. }
  48.  
  49. //自己定义的比较方法，用于sort
  50. bool myCompare( record &v1, record &v2){
  51. 	return v1.x < v2.x;
  52. }
  53.  
  54. //对MAXSIZE = 20个样本按X进行排序，这里直接调用自带sort函数，第三个参数是自己定义的比较方法（C++并不认识record,不知道怎么比较，我们要定义mycompare告诉他）
  55. void sortTrainingData(vector<record> &trainingData){
  56. 	sort(trainingData.begin(),trainingData.end(),myCompare);
  57. }
  58.  
  59. //给定输入集合和指定的hyphothesis计算对应的错误率
  60. double calculateError(vector<record> &trainingData,hyphothesis &h){
  61. 	int i;
  62. 	int error = 0;
  63. 	for(i = 0; i < DATASIZE; i++){
  64. 		int temp = h.s * sign(trainingData[i].x - h.theta);
  65. 		if(temp != trainingData[i].y)error++;
  66. 	}
  67. 	return error/double(DATASIZE);
  68. }
  69.  
  70. //由于S = 1 or -1,theta取值有21中，共有42中hyphothesis，我们计算得42种中最小的为E_in，并记录此时最小错误和对应的hyphothesis
  71. double E_in(vector<record> &trainingData,hyphothesis &bestH){
  72. 	hyphothesis temp;
  73. 	double min_errorRate = 1.0;
  74. 	int i;
  75. 	//s = 1时
  76. 	for(i = 0; i < DATASIZE+1; i++){
  77. 		temp.s = 1;
  78. 		if(i == 0)temp.theta = trainingData[0].x -1.0;//theta取值，theta小于最小
  79. 		else if(i == DATASIZE) temp.theta = trainingData[DATASIZE - 1].x + 1.0;//theta取值在两点之间
  80. 		else temp.theta = (trainingData[i-1].x + trainingData[i].x) / 2.0;//theta取值大于最大
  81.  
  82. 		double errorRate = calculateError(trainingData,temp);//如果此hyphothesis的错误更小，则替代
  83. 		if(errorRate < min_errorRate){
  84. 			bestH = temp;
  85. 			min_errorRate = errorRate;
  86. 		}
  87. 	}
  88. 	//s = -1时
  89. 	for(i = 0; i < DATASIZE+1; i++){
  90. 		temp.s = -1;
  91. 		if(i == 0)temp.theta = trainingData[0].x -1.0;
  92. 		else if(i == DATASIZE) temp.theta = trainingData[DATASIZE - 1].x + 1.0;
  93. 		else temp.theta = (trainingData[i-1].x + trainingData[i].x) / 2.0;
  94.  
  95. 		double errorRate = calculateError(trainingData,temp);
  96. 		if(errorRate < min_errorRate){
  97. 			bestH = temp;
  98. 			min_errorRate = errorRate;
  99. 		}
  100. 	}
  101. 	return min_errorRate;
  102. }
  103.  
  104. //利用16题得到的公式计算E_out，注意，浮点数求绝对值用fabs,而不是abs
  105. double E_out(hyphothesis &bestH){
  106. 	return 0.5 + 0.3 * double(bestH.s) * (double)(fabs(bestH.theta) - 1.0);
  107. }
  108.  
  109. void main(){
  110. 	int i;
  111. 	double totalE_inRate = 0.0;
  112. 	double totalE_outRate = 0.0;
  113.  
  114. 	int seed[5000];//由于要进行5000次求平均，所以要有5000个种子用于求随机数
  115. 	int j;
  116. 	for( j = 0; j < 5000; j++){
  117. 		seed[j] = rand();	//这5000个种子通过一次取随机数产生
  118. 	}
  119.  
  120. 	for(i = 0; i < 5000; i ++){
  121. 		srand(seed[i]);//每次取一个种子，那么每次产生的随机数序列就不一样了
  122. 	    vector<record> trainingData;
  123. 	    getRandData(trainingData);//随机生成训练样本
  124. 	    getNoise(trainingData);//加噪声
  125. 	    sortTrainingData(trainingData);//样本排序
  126. 		hyphothesis bestH = {0,0};
  127.         double min_errorRate = E_in(trainingData,bestH);//计算最优hyphothesis并记录最小错误率
  128. 	    cout<<"mininum E_in:"<<min_errorRate<<endl;
  129. 		totalE_inRate += min_errorRate;
  130. 		totalE_outRate += E_out(bestH);
  131. 		cout<<"E_out:"<<E_out(bestH)<<endl;//利用这次得到的最优hyphothesis求E_out
  132. 	}
  133. 	cout<<"average E_in:"<<totalE_inRate / 5000<<endl;//得平均E_in
  134. 	cout<<"E_out:"<<totalE_outRate / 5000<<endl;//得平均E_out
  135. }
